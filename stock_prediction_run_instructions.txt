import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

(Explain: pandas - data handling, numpy - math, matplotlib - visualization, sklearn - ML)

--------------------------------------------------------------
6Ô∏è‚É£ LOAD THE DATASET
--------------------------------------------------------------
data = pd.read_csv('stock_data.csv')
data.columns = data.columns.str.strip()
print(data.head())

(Explain: loads Google stock dataset and shows first 5 rows)

--------------------------------------------------------------
7Ô∏è‚É£ VISUALIZE THE STOCK TREND
--------------------------------------------------------------
plt.figure(figsize=(10,6))
plt.plot(data['Date'], data['Close'])
plt.title('Google Stock Price Over Time')
plt.xlabel('Date')
plt.ylabel('Close Price')
plt.show()

--------------------------------------------------------------
8Ô∏è‚É£ PREPARE DATA FOR PREDICTION
--------------------------------------------------------------
data['Days'] = np.arange(len(data))
X = data[['Days']]
y = data['Close']

--------------------------------------------------------------
9Ô∏è‚É£ TRAIN THE LINEAR REGRESSION MODEL
--------------------------------------------------------------
model = LinearRegression()
model.fit(X, y)

--------------------------------------------------------------
üîü PREDICT FUTURE PRICES
--------------------------------------------------------------
future_days = np.arange(len(data), len(data) + 10).reshape(-1, 1)
future_predictions = model.predict(future_days)

print("Next 10 Predicted Prices:")
print(future_predictions)

--------------------------------------------------------------
1Ô∏è‚É£1Ô∏è‚É£ VISUALIZE ACTUAL VS PREDICTED
--------------------------------------------------------------
plt.figure(figsize=(10,6))
plt.plot(data['Days'], y, label='Actual Price', color='blue')
plt.plot(future_days, future_predictions, label='Predicted Price', color='red', linestyle='--')
plt.title('Google Stock Price Prediction')
plt.xlabel('Days')
plt.ylabel('Stock Price')
plt.legend()
plt.show()

--------------------------------------------------------------
1Ô∏è‚É£2Ô∏è‚É£ SHOW MODEL ACCURACY (OPTIONAL)

from sklearn.metrics import r2_score
